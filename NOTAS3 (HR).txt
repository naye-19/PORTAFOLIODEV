      public String handlerResponseConsultarRecargasPaquetes(AppResponse jsonresponse, JsonObject requestSofia) throws Exception {

        int codigoStatus = jsonresponse.getResponse().getStatus();

        if (codigoStatus == Response.Status.OK.getStatusCode()) {

            String detalleXml = BuilderResponseTigo.crearResponseConsultarRecargaTigo(requestSofia, jsonresponse.getBody());

            XMLSofia xml = new XMLSofia(EnumsStatus.Status.OK.getStatus(), "CONSULTA EXITOSA", EnumsTipoMensajes.TipoMensajes.OK.getTipoMensaje());
            xml.crearResponseXmlSofia(requestSofia.getAsJsonObject("body"), detalleXml);

            return xml.getXML();

        }else if (codigoStatus == Response.Status.NO_CONTENT.getStatusCode()) {

            String detalleXml = "";

            XMLSofia xml = new XMLSofia(EnumsStatus.Status.ERROR.getStatus(), 
                    "Factura del cliente no encontrada en el sistema.",
                    EnumsTipoMensajes.TipoMensajes.ERROR.getTipoMensaje());
            
            xml.crearResponseXmlSofia(requestSofia.getAsJsonObject("body"), detalleXml);

            return xml.getXML();
        } else if (codigoStatus == EnumHttpStatus.HttpStatus.CLIENT_CLOSED_CONNECTION.get() || codigoStatus == EnumHttpStatus.HttpStatus.GATEWAY_TIMEOUT.get()||
            codigoStatus == EnumHttpStatus.HttpStatus.BAD_GATEWAY.get() || codigoStatus == Response.Status.REQUEST_TIMEOUT.getStatusCode()) {

            String detalleXml = "";

            XMLSofia xml = new XMLSofia(EnumsStatus.Status.ERROR.getStatus(),
                    "Error de comunicacion, Proveedor no respondio la peticion.",
                    EnumsTipoMensajes.TipoMensajes.ERROR.getTipoMensaje());

            xml.crearResponseXmlSofia(requestSofia.getAsJsonObject("body"), detalleXml);

            return xml.getXML();

        }else if (codigoStatus == EnumHttpStatus.HttpStatus.INTERNAL_SERVER_ERROR.get()) {

            String detalleXml = "";

            XMLSofia xml = new XMLSofia(EnumsStatus.Status.ERROR.getStatus(),
                    "Error interno reportar error al area de tecnologia.",
                    EnumsTipoMensajes.TipoMensajes.ERROR.getTipoMensaje());

            xml.crearResponseXmlSofia(requestSofia.getAsJsonObject("body"), detalleXml);

            return xml.getXML();    

        } else {

            String detalleXml = "";

            XMLSofia xml = new XMLSofia(EnumsStatus.Status.ERROR.getStatus(), 
                    ServiceCaracteres.limpiarCaracteresSofia(jsonresponse.getBody().get("errormessage").getAsString(),200),
                    EnumsTipoMensajes.TipoMensajes.ERROR.getTipoMensaje());
            
            xml.crearResponseXmlSofia(requestSofia.getAsJsonObject("body"), detalleXml);

            return xml.getXML();
        }

    }
