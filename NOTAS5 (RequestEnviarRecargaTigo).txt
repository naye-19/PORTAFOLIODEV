 *
 * @author MOMARTINEZ

public class RequestEnviarRecargaTigo {

    private final String codigoCajero;
    private final String codigoTransaccion;
    private final String numeroCelular;
    private final String tipoProductoInterno;

    private final String codigoProducto;
    private final String formaPago;

    private final String secuencial;

    private final BigDecimal montoChequeLocal;
    private final BigDecimal montoChequeNoLocal;
    private final BigDecimal montoChequeNuestro;

    private final BigDecimal montoTotal;
    private final BigDecimal montoComision;
    private final BigDecimal codigoReferencia;
    private final String codigoReferenciaOtro;

    public String getNombreTerminal() {
        return nombreTerminal;
    }

    public String getCodigoCajero() {
        return codigoCajero;
    }

    public String getCodigoTransaccion() {
        return codigoTransaccion;
    }

    public String getNumeroCelular() {
        return numeroCelular;
    }

    public String getTipoProductoInterno() {
        return tipoProductoInterno;
    }

    public String getCodigoSelector() {
        return codigoSelector;
    }

    public String getCodigoProducto() {
        return codigoProducto;
    }

    public String getFormaPago() {
        return formaPago;
    }

    public String getMoneda() {
        return moneda;
    }

    public String getSecuencial() {
        return secuencial;
    }

    public BigDecimal getMontoEfectivo() {
        return montoEfectivo;

        return montoTotal;

        this.moneda = moneda;
        this.secuencial = secuencial;
        this.montoEfectivo = montoEfectivo;
        this.montoAcumulado = montoAcumulado;
        this.montoChequeLocal = montoChequeLocal;
        this.montoChequeNoLocal = montoChequeNoLocal;
