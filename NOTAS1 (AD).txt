public class AdapterTigo {

    private static ServiceGeneric generic;

    public AdapterTigo() {
        this.generic = new ServiceGeneric(EnumServicios.Servicios.TIGO.getServicio());
    }

    public String adpOpcionesTigo(JsonObject request, Integer modalidad) throws IOException, Exception {

        JsonArray campos = UtilsSofia.getDetalleCampos(request);
        String opcionTigo = UtilsSofia.getValorCampoSinRemove(campos, "TIPTRANS2");

        if (opcionTigo.equals(EnumTipoProductosTigo.TipoProductos.RECARGAS.getTipoProducto()) && modalidad == 1) {

            return adpConsultarRecargas(request);

        } else if (opcionTigo.equals(EnumTipoProductosTigo.TipoProductos.RECARGAS.getTipoProducto()) && modalidad == 2) {

            return adpEnviarRecarga(request);
        } else if (opcionTigo.equals(EnumTipoProductosTigo.TipoProductos.PAQUETIGOS.getTipoProducto()) && modalidad == 1) {

            return adpConsultarPaqueTigos(request);
        } else if (opcionTigo.equals(EnumTipoProductosTigo.TipoProductos.PAQUETIGOS.getTipoProducto()) && modalidad == 2) {

            return adpEnviarPaqueTigo(request);
        } else {
            throw crearExceptionOpcionNoValida();
        }
    }

    public String adpConsultarRecargas(JsonObject request) throws IOException, Exception {

        Tracing tracing = new Tracing(generic, "adpConsultarRecargas");

        JsonObject encabezado = request.getAsJsonObject("body").getAsJsonObject("clsTransaccion");
        JsonArray campos = UtilsSofia.getDetalleCampos(request);

        RequestConsultarRecargaTigo requestTigo
                = new RequestConsultarRecargaTigo(UtilsSofia.getValorCampoSinRemove(campos, "NUMCEL"),
                        UtilsSofia.getValorCampoSinRemove(campos, "TIPTRANS2"),
                        UtilsSofia.getValorCampoSinRemove(campos, "MENTEXTO"),
                        encabezado.get("Cajero").getAsInt(),
                        encabezado.get("NumTransaccion").getAsInt(),
                        new BigDecimal(UtilsSofia.getValorCampoSinRemove(campos, "TNUDOCTRA"))
                );

        BuilderRequestTigo helperinternalTigo = new BuilderRequestTigo(this.generic);

        String response = helperinternalTigo.ConsultarRecargasTigo(requestTigo, request);

        tracing.Response_LOG_Tracing(generic, "adpConsultarRecargas");
        return response;
    }

    public String adpEnviarRecarga(JsonObject request) throws IOException, Exception {
        Tracing tracing = new Tracing(generic, "adpEnviarRecarga");

        if (!UtilsSofia.SoloUnaSeleccion(request, "SELTRAN")) {
            throw UtilsSofia.SoloUnaFacturaSeleccionada();
        }

        JsonObject encabezado = request.getAsJsonObject("body").getAsJsonObject("clsTransaccion");
        JsonArray campos = UtilsSofia.getDetalleCampos(request);

        JsonObject facturaSeleccionada = UtilsSofia.getFacturaSeleccionada(request, "SELTRAN");
        String descripcionTransaccion = "RECARGA TIGO DE: " + UtilsSofia.getValorCampoSinRemove(facturaSeleccionada.getAsJsonArray("clsColumna"), "MONTRN");

        RequestEnviarRecargaTigo requestTigo;
        requestTigo = new RequestEnviarRecargaTigo(UtilsSofia.getValorCampoSinRemove(campos, "MENTEXTO"),
                encabezado.get("Cajero").getAsString(),
                UtilsSofia.getValorCampoSinRemove(campos, "TDRTCD"),
                UtilsSofia.getValorCampoSinRemove(campos, "NUMCEL"),
                UtilsSofia.getValorCampoSinRemove(campos, "TIPTRANS2"),
                UtilsSofia.getValorCampoSinRemove(facturaSeleccionada.getAsJsonArray("clsColumna"), "SELECTOR"),
                UtilsSofia.getValorCampoSinRemove(facturaSeleccionada.getAsJsonArray("clsColumna"), "CODTRN"),
                UtilsSofia.getValorCampoSinRemove(campos, "FORPAG"),
                EnumsCore.TipoMonedas.LEMPIRAS.getCodigoMoneda(),
                UtilsSofia.getValorCampoSinRemove(campos, "SECUEN"),
                new BigDecimal(UtilsSofia.getValorCampoSinRemove(campos, "TVAEFEHOY")),
                new BigDecimal(UtilsSofia.getValorCampoSinRemove(campos, "MONTOACUM")),
                new BigDecimal(UtilsSofia.getValorCampoSinRemove(campos, "DOCNUESTROCARGO")),
                new BigDecimal(UtilsSofia.getValorCampoSinRemove(campos, "TVACHEHOY")),
                new BigDecimal(UtilsSofia.getValorCampoSinRemove(campos, "DOCNUESTROCARGO")),
                descripcionTransaccion,
                new BigDecimal(UtilsSofia.getValorCampoSinRemove(facturaSeleccionada.getAsJsonArray("clsColumna"), "MONTRN")),
                BigDecimal.ZERO,
                new BigDecimal(UtilsSofia.getValorCampoSinRemove(campos, "TNUDOCTRA")),
                UtilsSofia.getValorCampoSinRemove(campos, "NUMCEL"),
                encabezado.get("Agencia").getAsInt());

        BuilderRequestTigo helperinternalTigo = new BuilderRequestTigo(this.generic);

        String response = helperinternalTigo.EnviarRecarga(requestTigo, request);

        tracing.Response_LOG_Tracing(generic, "adpEnviarRecarga");
        return response;
    }
